#include <stdio.h>
#include <stdlib.h>

#define MAX_CITIES 100

struct Graph {
    int vertices;
    int adj[MAX_CITIES][MAX_CITIES];
};

void initializeGraph(struct Graph *G, int v) {
    G->vertices = v;
    for (int i = 0; i < v; i++)
        for (int j = 0; j < v; j++)
            G->adj[i][j] = 0;
}

void addEdge(struct Graph *G, int s, int d) {
    if (s >= 0 && s < G->vertices && d >= 0 && d < G->vertices)
        G->adj[s][d] = 1;
    else
        printf("Invalid edge: %d -> %d\n", s, d);
}

void DFS(struct Graph *G, int start, int visited[]) {
    printf("%d ", start);
    visited[start] = 1;

    for (int i = 0; i < G->vertices; i++)
        if (G->adj[start][i] && !visited[i])
            DFS(G, i, visited);
}

void BFS(struct Graph *G, int start) {
    int q[MAX_CITIES], front = 0, rear = 0;
    int visited[MAX_CITIES] = {0};

    printf("%d ", start);
    visited[start] = 1;
    q[rear++] = start;

    while (front < rear) {
        int cur = q[front++];
        for (int i = 0; i < G->vertices; i++) {
            if (G->adj[cur][i] && !visited[i]) {
                printf("%d ", i);
                visited[i] = 1;
                q[rear++] = i;
            }
        }
    }
}

int main() {
    struct Graph G;
    int n, edges, s, d, start;

    printf("Enter the number of cities: ");
    scanf("%d", &n);
    initializeGraph(&G, n);

    printf("Enter the number of edges: ");
    scanf("%d", &edges);

    for (int i = 0; i < edges; i++) {
        printf("Enter edge #%d (source destination): ", i + 1);
        scanf("%d %d", &s, &d);
        addEdge(&G, s, d);
    }

    printf("Enter the starting node for DFS/BFS: ");
    scanf("%d", &start);

    printf("DFS traversal starting from node %d: ", start);
    int visited[MAX_CITIES] = {0};
    DFS(&G, start, visited);
    printf("\n");

    printf("BFS traversal starting from node %d: ", start);
    BFS(&G, start);
    printf("\n");

    return 0;
}
